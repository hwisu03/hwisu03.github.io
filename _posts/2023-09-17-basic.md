#Dataframe의 변수를 이용하여 슬라이싱
df = pd.DataFrame({'A':[11,21,31], 'B':[12,22,32], 'C':[13,23,33]}, index=['1st','2nd','3rd'])

print(df.loc[lambda x: x.C<30])
print(df.loc[df['C']<30, ['A','B']])
